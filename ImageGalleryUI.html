     //Toggle between half and full width column sizes for posters (only on tablets and smaller)
     function resizeGallery() {
         if ($(window).width() < 992) {
             var element = document.getElementById("image1Container");
             element.classList.toggle("col-12");
             var element = document.getElementById("Image1");
             element.classList.toggle("left-column-realign");

             var element = document.getElementById("image2Container");
             element.classList.toggle("col-12");
             var element = document.getElementById("Image2");
             element.classList.toggle("right-column-realign");

             var element = document.getElementById("image3Container");
             element.classList.toggle("col-12");
             var element = document.getElementById("Image3");
             element.classList.toggle("left-column-realign");

             var element = document.getElementById("image4Container");
             element.classList.toggle("col-12");
             var element = document.getElementById("Image4");
             element.classList.toggle("right-column-realign");

             var element = document.getElementById("image5Container");
             element.classList.toggle("col-12");
             var element = document.getElementById("Image5");
             element.classList.toggle("left-column-realign");

             var element = document.getElementById("image6Container");
             element.classList.toggle("col-12");
             var element = document.getElementById("Image6");
             element.classList.toggle("right-column-realign");
         }
     }

     //Delays images linking to their own sections to adjust for resize
     function delay (URL) {
         if ($(window).width() < 992) {
             setTimeout( function() { window.location = URL }, 10 );
         }
     }

     //Removes info bubble when user taps it
     $(function removeBubble(){
         var $img       = $("#dialogueBubble")
           , $container = $("#dialogueContainer");
         $img.on("click", function removeBubble() {
             $img.remove();
             $container.removeClass().removeAttr("id");
         });
     });
