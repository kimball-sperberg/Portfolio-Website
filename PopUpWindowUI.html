  //Toggle [open] and [minimize] images
  function toggleOpenCloseSymbol() {
      var img1 = "Images/OButton.png",
         img2 = "Images/XButton.png";
      var imgElement = document.getElementById('xButton');
      imgElement.src = (imgElement.src === img1)? img2 : img1;
  }

  //Change Button Text onClick
  function toggleOpenCloseText() {
      var change = document.getElementById("toggle");
      if (change.innerHTML == "Close")
      {
          change.innerHTML = "Expand";
      }
      else {
          change.innerHTML = "Close";
      }
  }
